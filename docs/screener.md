This guide explains how to run market screens to obtain lists of securities. The `Screener.get()` method accepts the universes `"stocks"`, `"etfs"` and `"mutual_funds"`. The resulting `securityID` values can be used with the asset classes documented in [stock.md](stock.md), [etf.md](etf.md) and [fund.md](fund.md).

```python
import morningstarFetcher
from pprint import pprint

screener = morningstarFetcher.Screener() # Initialize the screener;

# Choose the universe: "mutual_funds", "etfs" or "stocks"
selected_investment_type = "mutual_funds"
markets_identifiers_list = screener.markets  # Display available markets;
print("Available Markets:")
pprint(markets_identifiers_list)

french_market_identifier = "fr"  # Define the market identifier (France);

ffields = screener.mutual_funds_fields  # Display fields for mutual funds;
print("\nMutual Funds Fields:")
pprint(ffields)

ffilter1 = ["maximumDrawdown[5y]", ">", "-10"] # 5Y Period Maximum Drawdown Filter;
ffilter2 = ["alpha[5yMonthly]", ">=", "2"] # 5Y Period Monthly Alpha Filter;
ffilter3 = ["totalReturn[5y]", ">=", "10"] # 5Y Period Total Return Filter;

ffilters = [ffilter1, ffilter2, ffilter3]  # Filter clause;
print("\nFilters:")
pprint(ffilters)

# Fetching mutual funds with the specified filters:
funds = screener.get(
    investment_type=selected_investment_type,
    market_id=french_market_identifier,
    sort_by="totalReturn[5y]:desc",
    filters=ffilters,
    pages=1,
)

total = len(funds)  # Total number of funds;
print("\nTotal Funds:")
pprint(total)  # Display the total number of funds;

columns = funds.columns.tolist()  # Columns;
print("\nColumns:")
pprint(columns)  # Display the columns;

fund_1 = funds.iloc[0].to_dict()  # First fund as a dictionary
print("\nFirst Fund:")
pprint(fund_1)

```

```python
# The same approach works for ETFs and Stocks
etfs = screener.get("etfs", "us", pages=1)
stocks = screener.get("stocks", "us", pages=1)
print(etfs.head(3))
print(stocks.head(3))
```
